function validSequenceSimple(braces){
	let stack = 0;
	for(let i = 0; i < braces.length; i++){
        if(braces[i] === '('){
            stack++;
        }
        if(braces[i] === ')'){
            stack--;
        }
        if(stack < 0){
            return false;
        }
    }
    if(stack === 0){
        return true;
    }
    else {
        return false;
    }
}
